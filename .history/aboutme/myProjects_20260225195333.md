## Rainfall Calculator
Calculates the total and average rainfall for the year. Also displays the lowest and highest month with rainfall in C# Visual Studios 2022.

```` C#
static void raininput(double[] drainfall)
{
    bool retval;
    double userinput = 0;

    for (int i = 0; i < drainfall.Length; i++)
    {
        
        Console.WriteLine("How much did it rain in " + MonthToWords(i) + " (in millimeters)");
        
        do
        {
            retval = double.TryParse(Console.ReadLine(), out userinput);
            if (!retval || userinput < 0)
            {
                Console.WriteLine("Invalid input! I need a number for "+MonthToWords(i)+" !!!!");
                i--;

                if (MonthToWords(i).Length == 0) // Code makes it stay in January(0) until user types in valid input and not go -1 and gives an error
                {
                    i = 0;
                }
            }
            

        }while (!retval);

        drainfall[i] = userinput;

        
    }
````
